	rnd=Math.exp(nrand()*arg["stdev"]*Math.sqrt(arg["time"])+arg["time"]*(arg["r"]-arg["stdev"]*arg["stdev"]/2.0))*arg["spot"];
	if(rnd>arg["strike"]) sum=sum+rnd-arg["strike"];
}
return ([100000,Math.exp(-arg["time"]*arg["r"])*sum/100000]);
